/**
 * 卡牌适配器 - 用于Jest测试
 * 
 * 此适配器封装了游戏中的卡牌系统，提供了一个与原始游戏代码行为一致的接口，
 * 便于在Jest测试环境中进行单元测试。
 */
 
// 导入依赖模块
const { v4: uuidv4 } = require('uuid');
const elementSystem = require('./element-system-adapter');
const skillEffects = require('./skill-effects-adapter');
 
/**
 * 创建一个新的卡牌
 * @param {string} name - 卡牌名称
 * @param {Object} stats - 卡牌属性
 * @param {Array} skills - 卡牌技能列表
 * @returns {Object} - 新创建的卡牌对象
 */
function createCard(name, stats, skills = []) {
    // 生成唯一ID
    const cardId = uuidv4();
    
    // 创建基础卡牌
    const card = {
        id: cardId,
        name: name,
        element: stats.element || elementSystem.ElementType.NEUTRAL,
        hp: stats.hp || 100,
        maxHp: stats.maxHp || stats.hp || 100,
        atk: stats.atk || 50,
        def: stats.def || 30,
        spd: stats.spd || 40,
        nrg: stats.nrg || 0,
        maxNrg: stats.maxNrg || 10,
        critRate: stats.critRate || 0.1,
        critDamage: stats.critDamage || 1.5,
        level: stats.level || 1,
        rarity: stats.rarity || 'common',
        skills: [],
        statusEffects: [],
        isActed: false
    };
    
    // 添加技能
    for (let i = 0; i < skills.length; i++) {
        card.skills.push(skills[i]);
    }
    
    return card;
}
 
/**
 * 创建一个玩家卡组
 * @param {number} count - 卡组中卡牌数量
 * @param {Object} options - 卡组选项
 * @returns {Array} - 卡组卡牌数组
 */
function createPlayerDeck(count = 3, options = {}) {
    const deck = [];
    
    // 根据要求生成指定数量的卡牌
    for (let i = 0; i < count; i++) {
        // 创建随机元素
        const elementTypes = Object.values(elementSystem.ElementType);
        const randomElement = elementTypes[Math.floor(Math.random() * elementTypes.length)];
        
        // 创建基础技能
        const basicAttack = skillEffects.createSkill('基础攻击', skillEffects.SkillType.DAMAGE, {
            multiplier: 1.0,
            element: randomElement,
            energyCost: 1,
            description: '对敌人进行基础攻击'
        });
        
        // 创建卡牌
        const cardName = `玩家卡牌 ${i + 1}`;
        const cardStats = {
            element: randomElement,
            hp: 100 + (i * 20),
            maxHp: 100 + (i * 20),
            atk: 50 + (i * 10),
            def: 30 + (i * 5),
            spd: 40 + (i * 5),
            nrg: 5,
            maxNrg: 10,
            critRate: 0.1 + (i * 0.05),
            critDamage: 1.5,
            level: options.level || 1,
            rarity: options.rarity || 'common'
        };
        
        // 添加特殊技能
        const skills = [basicAttack];
        
        if (i === 0) {
            // 为第一张卡牌添加伤害技能
            const damageSkill = skillEffects.createSkill('元素冲击', skillEffects.SkillType.DAMAGE, {
                multiplier: 1.5,
                element: randomElement,
                energyCost: 3,
                cooldown: 2,
                description: `造成${randomElement}元素伤害`
            });
            skills.push(damageSkill);
        } else if (i === 1) {
            // 为第二张卡牌添加增益技能
            const buffSkill = skillEffects.createSkill('力量增强', skillEffects.SkillType.BUFF, {
                energyCost: 2,
                targetType: skillEffects.TargetType.SELF,
                statusEffect: {
                    type: skillEffects.StatusEffectType.ATK_UP,
                    value: 15,
                    duration: 3
                },
                description: '增加自身攻击力'
            });
            skills.push(buffSkill);
        } else {
            // 为其他卡牌添加群体伤害技能
            const aoeSkill = skillEffects.createSkill('元素风暴', skillEffects.SkillType.DAMAGE, {
                multiplier: 0.8,
                element: randomElement,
                energyCost: 4,
                cooldown: 3,
                targetType: skillEffects.TargetType.ALL,
                description: `对所有敌人造成${randomElement}元素伤害`
            });
            skills.push(aoeSkill);
        }
        
        const card = createCard(cardName, cardStats, skills);
        deck.push(card);
    }
    
    return deck;
}
 
/**

/**
 * 复制卡牌
 * @param {Object} card - 源卡牌
 * @returns {Object} - 卡牌副本
 */
function cloneCard(card) {
    if (!card) return null;
    
    // 创建基础卡牌副本
    const clone = { ...card };
    
    // 复制技能
    clone.skills = [];
    for (let i = 0; i < card.skills.length; i++) {
        clone.skills.push({ ...card.skills[i] });
    }
    
    // 复制状态效果
    clone.statusEffects = [];
    for (let i = 0; i < card.statusEffects.length; i++) {
        clone.statusEffects.push({ ...card.statusEffects[i] });
    }
    
    return clone;
}
 
/**
 * 重置卡牌回合状态
 * @param {Object} card - 需要重置的卡牌
 */
function resetCardTurnState(card) {
    card.isActed = false;
    
    // 重置技能冷却
    for (let i = 0; i < card.skills.length; i++) {
        if (card.skills[i].currentCooldown > 0) {
            card.skills[i].currentCooldown--;
        }
    }
}
 
/**
 * 恢复卡牌能量
 * @param {Object} card - 需要恢复能量的卡牌
 * @param {number} amount - 恢复量，默认为2
 */
function restoreCardEnergy(card, amount = 2) {
    card.nrg = Math.min(card.maxNrg, card.nrg + amount);
}
 
/**
 * 为卡牌应用伤害
 * @param {Object} targetCard - 目标卡牌
 * @param {number} damage - 伤害值
 * @returns {Object} - 伤害结果
 */
function applyDamage(targetCard, damage) {
    const previousHp = targetCard.hp;
    targetCard.hp = Math.max(0, targetCard.hp - damage);
    
    return {
        target: targetCard.id,
        damage: damage,
        previousHp: previousHp,
        currentHp: targetCard.hp,
        isDead: targetCard.hp <= 0
    };
}
 
/**
 * 为卡牌应用治疗
 * @param {Object} targetCard - 目标卡牌
 * @param {number} healing - 治疗量
 * @returns {Object} - 治疗结果
 */
function applyHealing(targetCard, healing) {
    const previousHp = targetCard.hp;
    const maxHeal = targetCard.maxHp - targetCard.hp;
    const actualHeal = Math.min(healing, maxHeal);
    
    targetCard.hp += actualHeal;
    
    return {
        target: targetCard.id,
        healing: actualHeal,
        previousHp: previousHp,
        currentHp: targetCard.hp
    };
}
 
/**
 * 检查卡牌是否死亡
 * @param {Object} card - 卡牌对象
 * @returns {boolean} - 是否死亡
 */
function isCardDead(card) {
    return card.hp <= 0;
}
 
/**
 * 获取卡牌技能消耗
 * @param {Object} card - 卡牌对象
 * @param {Object} skill - 技能对象
 * @returns {number} - 实际消耗能量
 */
function getSkillCost(card, skill) {
    // 基础消耗
    let cost = skill.energyCost || 0;
    
    // 如果卡牌有减少能量消耗的状态效果，可以在这里应用
    // 这里仅作为示例
    
    return Math.max(0, cost);
}
 
/**
 * 使用卡牌技能
 * @param {Object} card - 施放技能的卡牌
 * @param {Object} skill - 要使用的技能
 * @param {Array} targets - 技能目标
 * @returns {Object} - 技能使用结果
 */
function useCardSkill(card, skill, targets) {
    // 检查能量是否足够
    const cost = getSkillCost(card, skill);
    if (card.nrg < cost) {
        return {
            success: false,
            message: '能量不足'
        };
    }
    
    // 检查技能是否在冷却中
    if (skill.currentCooldown > 0) {
        return {
            success: false,
            message: '技能冷却中'
        };
    }
    
    // 消耗能量
    card.nrg -= cost;
    
    // 设置技能冷却
    if (skill.cooldown > 0) {
        skill.currentCooldown = skill.cooldown;
    }
    
    // 标记卡牌已行动
    card.isActed = true;
    
    // 根据技能类型处理效果
    let result = {
        success: true,
        caster: card.id,
        skill: skill.name,
        type: skill.type,
        targets: []
    };
    
    // 单体技能
    if (skill.targetType === skillEffects.TargetType.SINGLE && targets.length > 0) {
        const target = targets[0];
        
        switch (skill.type) {
            case skillEffects.SkillType.DAMAGE:
                const damageResult = skillEffects.calculateDamageSkill(card, target, skill);
                applyDamage(target, damageResult.value);
                result.targets.push({
                    id: target.id,
                    effect: 'damage',
                    value: damageResult.value,
                    isCritical: damageResult.isCritical,
                    isDodged: damageResult.isDodged
                });
                break;
                
            case skillEffects.SkillType.HEAL:
                const healResult = skillEffects.calculateHealSkill(card, target, skill);
                applyHealing(target, healResult.value);
                result.targets.push({
                    id: target.id,
                    effect: 'heal',
                    value: healResult.value
                });
                break;
                
            case skillEffects.SkillType.BUFF:
            case skillEffects.SkillType.DEBUFF:
                const statusResult = skillEffects.applyStatusEffect(card, target, skill);
                result.targets.push({
                    id: target.id,
                    effect: 'status',
                    statusType: skill.statusEffect.type,
                    success: statusResult.success
                });
                break;
        }
    } 
    // 群体技能
    else if (skill.targetType === skillEffects.TargetType.ALL && targets.length > 0) {
        const aoeResult = skillEffects.applyAOESkill(card, targets, skill);
        
        // 将结果添加到返回对象
        for (let i = 0; i < aoeResult.results.length; i++) {
            const targetResult = aoeResult.results[i];
            const target = targets[i];
            
            if (skill.type === skillEffects.SkillType.DAMAGE) {
                applyDamage(target, targetResult.value);
                result.targets.push({
                    id: target.id,
                    effect: 'damage',
                    value: targetResult.value,
                    isCritical: targetResult.isCritical,
                    isDodged: targetResult.isDodged
                });
            } else if (skill.type === skillEffects.SkillType.BUFF || skill.type === skillEffects.SkillType.DEBUFF) {
                result.targets.push({
                    id: target.id,
                    effect: 'status',
                    statusType: skill.statusEffect.type,
                    success: targetResult.success
                });
            }
        }
    }
    
    return result;
}
